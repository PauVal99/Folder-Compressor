470  private void manualOptions(String alg)

472  if(sourceCompress.isFile()){

475  errorMessage("Choose a quality beetwen [0..100]");
476  else runCompressor("JPEG", Integer.parseInt(selectQual.getText()));

486  errorMessage("Choose a quality beetwen [0..100]");
487  else runCompressor(alg, Integer.parseInt(selectQual.getText()));
